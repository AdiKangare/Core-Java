class thread1 extends Thread
{
	public void run()
	{
		for(char ch='A';ch<='Z';ch++)
		{
			System.out.println(ch);
			try
			{
				Thread.sleep(2000);
			}catch(InterruptedException e)
			{
				System.out.println(e);
			}
		}
	}
}
public class thread 
{

	public static void main(String[] args)
	{
		thread1 t1=new thread1();
		t1.start();
	}

}
